### 1.2 核心设计原则

| 原则 | 说明 |
|------|------|
| **规则引擎优先** | 所有 pass/fail 判断由规则引擎完成，不用 AI |
| **解析器独立** | 每个 parser 只负责提取结构化数据，不做判断 |
| **AI 最小化** | AI 只做 "解释 + 修复建议"，且可选（离线模式跳过） |
| **无副作用** | 工具只读取文件，永远不修改任何设计文件 |
| **单文件可运行** | MVP 不需要 Docker/Redis/Nginx，`pip install` 即用 |

## 2. 核心模块设计

### 2.1 规则引擎 (engine.py)

规则引擎是整个工具的核心。它负责：
1. 接收解析器产出的结构化数据 (Findings)
2. 匹配规则，生成检查结果 (Issues)
3. 汇总评分
4. 触发 AI 解读（可选）

```python
# core/engine.py

from dataclasses import dataclass, field
from enum import Enum
from typing import List, Optional
from pathlib import Path

from vivado_ai.core.rules.base import Rule, RuleResult
from vivado_ai.core.rules.registry import RuleRegistry
from vivado_ai.core.parsers.report_parser import ReportParser
from vivado_ai.core.parsers.log_parser import LogParser
from vivado_ai.core.parsers.xdc_parser import XDCParser
from vivado_ai.core.scorer import ComplianceScorer
from vivado_ai.core.ai_interpreter import AIInterpreter
from vivado_ai.models.finding import Findings
from vivado_ai.models.issue import Issue, Severity


class CheckMode(Enum):
    LINT = "lint"
    CHECK = "check"
    ANALYZE = "analyze"


@dataclass
class CheckConfig:
    mode: CheckMode
    rtl_dir: Optional[Path] = None
    xdc_files: List[Path] = field(default_factory=list)
    reports_dir: Optional[Path] = None
    log_dir: Optional[Path] = None
    part: Optional[str] = None
    enable_ai: bool = True
    rule_groups: List[str] = field(default_factory=lambda: ["all"])


@dataclass
class CheckResult:
    issues: List[Issue]
    score: int  # 0-100
    summary: dict


class MethodologyEngine:
    """
    Vivado Methodology 检查引擎

    工作流程:
    1. 解析输入数据 → Findings
    2. 加载规则 → 匹配 Findings → Issues
    3. 汇总评分
    4. (可选) AI 解读
    """

    def __init__(self, config: CheckConfig):
        self.config = config
        self.registry = RuleRegistry()
        self.scorer = ComplianceScorer()
        self.ai = AIInterpreter() if config.enable_ai else None

        # 解析器
        self.report_parser = ReportParser()
        self.log_parser = LogParser()
        self.xdc_parser = XDCParser()

    def run(self) -> CheckResult:
        """执行完整检查流程"""
        # Step 1: 收集数据
        findings = self._collect_findings()

        # Step 2: 加载并执行规则
        rules = self.registry.get_rules(
            mode=self.config.mode,
            groups=self.config.rule_groups
        )
        issues = []
        for rule in rules:
            result = rule.check(findings)
            issues.extend(result.issues)

        # Step 3: AI 解读 (可选)
        if self.ai:
            for issue in issues:
                if issue.severity in (Severity.FAIL, Severity.CRITICAL):
                    issue.ai_explanation = self.ai.explain(issue)

        # Step 4: 评分
        score = self.scorer.score(issues)

        return CheckResult(
            issues=issues,
            score=score,
            summary=self._build_summary(issues, score)
        )

    def _collect_findings(self) -> Findings:
        """根据模式收集数据"""
        findings = Findings()

        if self.config.mode == CheckMode.LINT:
            # 解析 XDC 文件
            for xdc in self.config.xdc_files:
                findings.xdc_data = self.xdc_parser.parse(xdc)
            # Phase 2: 解析 RTL

        elif self.config.mode == CheckMode.CHECK:
            # 解析 Vivado 报告文件
            if self.config.reports_dir:
                findings = self.report_parser.parse_dir(
                    self.config.reports_dir
                )

        elif self.config.mode == CheckMode.ANALYZE:
            # 解析 log 文件
            if self.config.log_dir:
                findings = self.log_parser.parse_dir(
                    self.config.log_dir
                )

        return findings

    def _build_summary(self, issues: List[Issue], score: int) -> dict:
        by_severity = {}
        for issue in issues:
            by_severity[issue.severity.value] = (
                by_severity.get(issue.severity.value, 0) + 1
            )
        return {
            "total_issues": len(issues),
            "by_severity": by_severity,
            "score": score,
        }
```

### 2.2 合规评分器 (scorer.py)

```python
# core/scorer.py

from vivado_ai.models.issue import Issue, Severity


class ComplianceScorer:
    """
    方法论合规评分 (0-100)

    扣分规则:
    - CRITICAL: -15 分/条
    - FAIL: -10 分/条
    - WARN: -3 分/条
    - INFO: 不扣分
    """

    PENALTY = {
        Severity.CRITICAL: 15,
        Severity.FAIL: 10,
        Severity.WARN: 3,
        Severity.INFO: 0,
    }

    def score(self, issues: list[Issue]) -> int:
        penalty = sum(self.PENALTY[i.severity] for i in issues)
        return max(0, 100 - penalty)
```

### 3.1 规则基类 (rules/base.py)

```python
# core/rules/base.py

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import List, Optional

from vivado_ai.models.finding import Findings
from vivado_ai.models.issue import Issue, Severity


@dataclass
class RuleResult:
    """规则执行结果"""
    rule_id: str
    issues: List[Issue] = field(default_factory=list)


class Rule(ABC):
    """
    方法论检查规则基类

    每条规则对应 UG949/UG1292 中的一个检查项。
    规则只做确定性判断，不做 AI 推理。
    """

    # 规则元数据（子类必须定义）
    id: str = ""               # e.g. "CONST-001"
    name: str = ""             # e.g. "Clock Period Constraints"
    group: str = ""            # e.g. "A" (约束), "E" (综合Log)
    severity: Severity = Severity.FAIL
    ug949_ref: str = ""        # e.g. "Ch5: Baseline Constraints"
    ug1292_ref: str = ""       # e.g. "Reducing Logic Delay" (可选)
    applicable_modes: List[str] = field(default_factory=lambda: ["check"])

    @abstractmethod
    def check(self, findings: Findings) -> RuleResult:
        """
        执行检查

        Args:
            findings: 解析器提取的结构化数据

        Returns:
            RuleResult: 包含发现的问题列表（空列表 = PASS）
        """
        pass

    def _create_issue(
        self,
        message: str,
        detail: str = "",
        fix_suggestion: str = "",
        location: str = "",
    ) -> Issue:
        return Issue(
            rule_id=self.id,
            rule_name=self.name,
            severity=self.severity,
            message=message,
            detail=detail,
            fix_suggestion=fix_suggestion,
            location=location,
            ug949_ref=self.ug949_ref,
            ug1292_ref=self.ug1292_ref,
        )
```

### 3.2 规则注册表 (rules/registry.py)

```python
# core/rules/registry.py

from typing import List, Dict, Type
from vivado_ai.core.rules.base import Rule


class RuleRegistry:
    """规则注册表 — 管理所有可用规则"""

    _rules: Dict[str, Type[Rule]] = {}

    @classmethod
    def register(cls, rule_class: Type[Rule]):
        """注册规则"""
        cls._rules[rule_class.id] = rule_class
        return rule_class

    def get_rules(
        self,
        mode: str = "all",
        groups: List[str] = None,
    ) -> List[Rule]:
        """获取适用于当前模式的规则实例"""
        rules = []
        for rule_cls in self._rules.values():
            rule = rule_cls()
            # 过滤模式
            if mode != "all" and mode not in rule.applicable_modes:
                continue
            # 过滤规则组
            if groups and "all" not in groups:
                if rule.group not in groups:
                    continue
            rules.append(rule)
        return rules

    def list_rules(self) -> List[dict]:
        """列出所有已注册规则"""
        return [
            {
                "id": cls.id,
                "name": cls.name,
                "group": cls.group,
                "severity": cls.severity.value,
                "modes": cls.applicable_modes,
            }
            for cls in self._rules.values()
        ]
```

### 3.3 规则实现示例

#### 约束规则 (rules/constraint_rules.py)

```python
# core/rules/constraint_rules.py

import re
from vivado_ai.core.rules.base import Rule, RuleResult
from vivado_ai.core.rules.registry import RuleRegistry
from vivado_ai.models.finding import Findings
from vivado_ai.models.issue import Severity


@RuleRegistry.register
class ClockPeriodConstraint(Rule):
    """CONST-001: 检查所有时钟是否有周期约束"""
    id = "CONST-001"
    name = "Clock Period Constraints"
    group = "A"
    severity = Severity.FAIL
    ug949_ref = "Ch5: Establishing a Baseline"
    applicable_modes = ["lint", "check"]

    def check(self, findings: Findings) -> RuleResult:
        issues = []

        # 从 XDC 数据中提取已约束的时钟
        constrained_clocks = set()
        if findings.xdc_data:
            for cmd in findings.xdc_data.commands:
                if cmd.type == "create_clock":
                    constrained_clocks.add(cmd.args.get("name", ""))

        # 从 report_clock_networks 中提取所有时钟网络
        if findings.clock_networks:
            for net in findings.clock_networks:
                if net.name not in constrained_clocks:
                    issues.append(self._create_issue(
                        message=f"Clock '{net.name}' has no period constraint",
                        detail=f"{net.endpoint_count} endpoints driven by "
                               f"unconstrained clock",
                        fix_suggestion=(
                            f"create_clock -period <period> -name {net.name} "
                            f"[get_ports {net.source_port}]"
                        ),
                        location=net.source_port,
                    ))

        # Lint 模式: 检查 XDC 文件中的 create_clock 覆盖情况
        if findings.xdc_data and not findings.clock_networks:
            ports_with_clock_names = [
                p for p in findings.xdc_data.ports
                if any(kw in p.name.lower()
                       for kw in ["clk", "clock", "ref_clk"])
            ]
            for port in ports_with_clock_names:
                if port.name not in constrained_clocks:
                    issues.append(self._create_issue(
                        message=f"Port '{port.name}' looks like a clock "
                                f"but has no create_clock",
                        fix_suggestion=(
                            f"create_clock -period <period> "
                            f"-name {port.name} "
                            f"[get_ports {port.name}]"
                        ),
                        location=str(port.file_path),
                    ))

        return RuleResult(rule_id=self.id, issues=issues)


@RuleRegistry.register
class ClockInteractionConstraint(Rule):
    """CONST-004: 检查时钟域交互是否声明"""
    id = "CONST-004"
    name = "Clock Domain Interaction Constraints"
    group = "A"
    severity = Severity.FAIL
    ug949_ref = "Ch5: CDC Constraints"
    applicable_modes = ["lint", "check"]

    def check(self, findings: Findings) -> RuleResult:
        issues = []

        # 从 report_clock_interaction 中检查 unsafe 交互
        if findings.clock_interactions:
            for interaction in findings.clock_interactions:
                if interaction.inter_class in ("unsafe", "no"):
                    issues.append(self._create_issue(
                        message=(
                            f"Unsafe clock interaction: "
                            f"{interaction.from_clock} → "
                            f"{interaction.to_clock} "
                            f"(class: {interaction.inter_class})"
                        ),
                        fix_suggestion=(
                            f"set_clock_groups -asynchronous "
                            f"-group [get_clocks {interaction.from_clock}] "
                            f"-group [get_clocks {interaction.to_clock}]"
                        ),
                    ))

        return RuleResult(rule_id=self.id, issues=issues)
```

```python
# core/rules/synth_rules.py

from vivado_ai.core.rules.base import Rule, RuleResult
from vivado_ai.core.rules.registry import RuleRegistry
from vivado_ai.models.finding import Findings
from vivado_ai.models.issue import Severity


@RuleRegistry.register
class NonDedicatedClockRoute(Rule):
    """SYNTH-006: 非专用时钟布线"""
    id = "SYNTH-006"
    name = "Non-Dedicated Clock Route"
    group = "E"
    severity = Severity.FAIL
    ug949_ref = "Ch3: CLOCK_DEDICATED_ROUTE"
    applicable_modes = ["analyze"]

    # Vivado 消息码匹配
    TARGET_CODES = [
        "Synth 8-524",       # non-dedicated clock route
        "Vivado 12-4737",    # CLOCK_DEDICATED_ROUTE warning
        "DRC RTSTAT-1",      # routing not using dedicated resource
    ]

    def check(self, findings: Findings) -> RuleResult:
        issues = []
        if not findings.synth_log:
            return RuleResult(rule_id=self.id)

        for msg in findings.synth_log.messages:
            if msg.code in self.TARGET_CODES:
                issues.append(self._create_issue(
                    message=f"Non-dedicated clock route: {msg.text}",
                    detail=f"Code: {msg.code}",
                    fix_suggestion=(
                        "1. Use GCIO (Global Clock I/O) pins\n"
                        "2. Add: set_property CLOCK_DEDICATED_ROUTE "
                        "TRUE [get_nets <clk_net>]\n"
                        "3. Ensure clock goes through MMCM/PLL/BUFG"
                    ),
                    location=msg.source,
                ))

        return RuleResult(rule_id=self.id, issues=issues)


@RuleRegistry.register
class UnintendedLatch(Rule):
    """SYNTH-002: 意外 Latch 生成"""
    id = "SYNTH-002"
    name = "Unintended Latch Generation"
    group = "E"
```

```python
# core/parsers/report_parser.py

import re
from pathlib import Path
from dataclasses import dataclass, field
from typing import List, Optional

from vivado_ai.models.finding import Findings


@dataclass
class TimingPath:
    """时序路径"""
    slack: float
    start_point: str
    end_point: str
    path_type: str  # "setup" | "hold"
    requirement: float
    datapath_delay: float
    logic_delay: float
    net_delay: float
    clock_skew: float
    logic_levels: int


@dataclass
class TimingSummary:
    """时序汇总"""
    wns: float
    tns: float
    whs: float
    ths: float
    failing_endpoints: int
    total_endpoints: int


@dataclass
class MethodologyCheck:
    """report_methodology 单条检查"""
    check_id: str       # e.g. "TIMING-14"
    severity: str       # "CRITICAL WARNING" | "WARNING" | "INFO"
    message: str
    details: List[str] = field(default_factory=list)


@dataclass
class ClockInteraction:
    """时钟域交互"""
    from_clock: str
    to_clock: str
    inter_class: str   # "safe" | "no" | "unsafe"
    wns: Optional[float] = None


class ReportParser:
    """
    解析 Vivado 报告文件 (.rpt)

    支持的报告类型:
    - report_timing_summary 输出
    - report_methodology 输出
    - report_clock_interaction 输出
    - report_failfast 输出
    - report_design_analysis 输出
    - report_clock_networks 输出
    """

    def parse_dir(self, reports_dir: Path) -> Findings:
        """解析报告目录下的所有 .rpt 文件"""
        findings = Findings()
        rpt_files = list(reports_dir.glob("**/*.rpt"))

        for rpt_file in rpt_files:
            content = rpt_file.read_text(encoding="utf-8", errors="ignore")
            name_lower = rpt_file.name.lower()

            if "timing_summary" in name_lower or "timing" in name_lower:
                findings.timing_summary = self._parse_timing_summary(content)
                findings.timing_paths = self._parse_timing_paths(content)
            elif "methodology" in name_lower:
                findings.methodology_checks = (
                    self._parse_methodology(content)
                )
            elif "clock_interaction" in name_lower:
                findings.clock_interactions = (
                    self._parse_clock_interaction(content)
                )
            elif "clock_network" in name_lower:
                findings.clock_networks = (
                    self._parse_clock_networks(content)
                )
            elif "failfast" in name_lower:
                findings.failfast_data = self._parse_failfast(content)
            elif "utilization" in name_lower:
                findings.utilization = self._parse_utilization(content)

        return findings

    def _parse_timing_summary(self, content: str) -> TimingSummary:
        """解析 timing summary"""
        wns = self._extract_float(r"WNS\s*:\s*([-\d.]+)", content)
        tns = self._extract_float(r"TNS\s*:\s*([-\d.]+)", content)
        whs = self._extract_float(r"WHS\s*:\s*([-\d.]+)", content)
        ths = self._extract_float(r"THS\s*:\s*([-\d.]+)", content)
        failing = self._extract_int(
            r"Failing Endpoints\s*:\s*(\d+)", content
        )
        total = self._extract_int(
            r"Total Endpoints\s*:\s*(\d+)", content
        )
        return TimingSummary(
            wns=wns or 0, tns=tns or 0,
            whs=whs or 0, ths=ths or 0,
            failing_endpoints=failing or 0,
            total_endpoints=total or 0,
        )

    def _parse_timing_paths(self, content: str) -> List[TimingPath]:
        """解析关键时序路径"""
        # Vivado timing report 的路径格式较固定
        paths = []
        # 简化实现 — 实际需要更精细的解析
        path_pattern = re.compile(
            r"Slack\s*\(.*?\)\s*:\s*([-\d.]+)\s*ns.*?"
            r"Source:\s*(\S+).*?"
            r"Destination:\s*(\S+).*?"
            r"Requirement:\s*([-\d.]+)\s*ns.*?"
            r"Data Path Delay:\s*([-\d.]+)\s*ns.*?"
            r"Logic Delay:\s*([-\d.]+)\s*ns.*?"
            r"Net Delay:\s*([-\d.]+)\s*ns.*?"
            r"Clock Skew:\s*([-\d.]+)\s*ns",
            re.DOTALL
        )
        for m in path_pattern.finditer(content):
            paths.append(TimingPath(
                slack=float(m.group(1)),
                start_point=m.group(2),
                end_point=m.group(3),
                path_type="setup",
                requirement=float(m.group(4)),
                datapath_delay=float(m.group(5)),
                logic_delay=float(m.group(6)),
                net_delay=float(m.group(7)),
                clock_skew=float(m.group(8)),
                logic_levels=0,  # 需要额外解析
            ))
        return paths

    def _parse_methodology(self, content: str) -> List[MethodologyCheck]:
        """解析 report_methodology 输出"""
        checks = []
        # 匹配 TIMING-*, XDC-* 等检查项
        check_pattern = re.compile(
            r"\|\s*(TIMING-\d+|XDC\w*-\d+)\s*\|"
            r"\s*(CRITICAL WARNING|WARNING|INFO)\s*\|"
            r"\s*(.*?)\s*\|",
            re.DOTALL
        )
        for m in check_pattern.finditer(content):
            checks.append(MethodologyCheck(
                check_id=m.group(1),
                severity=m.group(2),
                message=m.group(3).strip(),
            ))
        return checks

    def _parse_clock_interaction(
        self, content: str
    ) -> List[ClockInteraction]:
        """解析 report_clock_interaction 输出"""
        interactions = []
        # 匹配时钟交互矩阵
        inter_pattern = re.compile(
            r"(\S+)\s*→\s*(\S+)\s*\|\s*(\w+)\s*"
            r"(?:\|\s*([-\d.]+))?",
            re.MULTILINE
        )
        for m in inter_pattern.finditer(content):
            interactions.append(ClockInteraction(
                from_clock=m.group(1),
                to_clock=m.group(2),
                inter_class=m.group(3),
                wns=float(m.group(4)) if m.group(4) else None,
            ))
        return interactions

    # ... 更多解析方法

    @staticmethod
    def _extract_float(pattern: str, text: str) -> Optional[float]:
        m = re.search(pattern, text)
        return float(m.group(1)) if m else None

    @staticmethod
    def _extract_int(pattern: str, text: str) -> Optional[int]:
        m = re.search(pattern, text)
        return int(m.group(1)) if m else None
```

### 4.2 Log 解析器 (parsers/log_parser.py)

```python
# core/parsers/log_parser.py

import re
from pathlib import Path
from dataclasses import dataclass, field
from typing import List, Optional

from vivado_ai.models.finding import Findings


@dataclass
class LogMessage:
    """Log 中的单条消息"""
    level: str           # "ERROR" | "CRITICAL WARNING" | "WARNING" | "INFO"
    code: str            # e.g. "Synth 8-327", "Place 30-487"
    text: str            # 消息内容
    source: str = ""     # 来源文件/模块
    timestamp: str = ""


@dataclass
class CongestionReport:
    """拥塞报告"""
    level: int            # 1-5
    region: str           # e.g. "X12Y5:W16xH16"
    top_modules: List[str] = field(default_factory=list)


@dataclass
class StageLog:
    """单个阶段的 Log 数据"""
    stage: str            # "synthesis" | "opt" | "place" | "phys_opt" | "route"
    messages: List[LogMessage] = field(default_factory=list)
    congestion_reports: List[CongestionReport] = field(default_factory=list)
    wns_before_phys_opt: Optional[float] = None
    wns_after_phys_opt: Optional[float] = None
    duration_seconds: Optional[float] = None
    resource_summary: dict = field(default_factory=dict)


class LogParser:
    """
    解析 Vivado 编译 log 文件

    支持:
    - Vivado 综合日志
    - opt_design 日志
    - place_design 日志
    - phys_opt_design 日志
    - route_design 日志
    - 完整编译日志 (包含所有阶段)
    """

    # Vivado 消息级别和编号的正则模式
    MESSAGE_PATTERN = re.compile(
        r"(ERROR|CRITICAL WARNING|WARNING|INFO)\s*:\s*"
        r"(?:\[(\S+?)\]\s*)?"     # 消息码, e.g. [Synth 8-327]
        r"(.*?)\s*$",
        re.MULTILINE
    )

    # 阶段标识符
    STAGE_MARKERS = {
        "synthesis": [
            "Starting Synthesis",
            "synth_design",
            "Running synth_design",
        ],
        "opt": [
            "Starting opt_design",
            "Running opt_design",
        ],
        "place": [
            "Starting Placement",
            "Running place_design",
        ],
        "phys_opt": [
            "Starting Physical Optimization",
            "Running phys_opt_design",
        ],
        "route": [
            "Starting Routing",
            "Running route_design",
        ],
    }

    # WNS 提取模式
    TIMING_SUMMARY_PATTERN = re.compile(
        r"Timing Summary\s*\|.*?"
        r"WNS\s*=\s*([-\d.]+).*?"
        r"TNS\s*=\s*([-\d.]+)",
        re.DOTALL
    )

    # 拥塞报告模式
    CONGESTION_PATTERN = re.compile(
        r"Congestion\s*(?:level|Level)\s*(\d+).*?"
        r"(?:Region|Window)\s*[:=]\s*(\S+)",
        re.DOTALL
    )

    def parse_dir(self, log_dir: Path) -> Findings:
        """解析目录下的所有 log 文件"""
        findings = Findings()
        log_files = list(log_dir.glob("**/*.log"))

        for log_file in log_files:
            content = log_file.read_text(encoding="utf-8", errors="ignore")
            name_lower = log_file.name.lower()

            if "synth" in name_lower or "runme" in name_lower:
                findings.synth_log = self._parse_stage(content, "synthesis")
            elif "place" in name_lower:
                findings.place_log = self._parse_stage(content, "place")
            elif "route" in name_lower:
                findings.route_log = self._parse_stage(content, "route")
            else:
                # 可能是完整编译 log，尝试拆分
                self._parse_full_log(content, findings)

        return findings

    def _parse_stage(self, content: str, stage: str) -> StageLog:
        """解析单个阶段的 log"""
        log = StageLog(stage=stage)

        # 提取所有消息
        log.messages = self._extract_messages(content)

        # 提取拥塞报告
        log.congestion_reports = self._extract_congestion(content)

        # 提取 WNS
        wns_values = self._extract_all_wns(content)
        if len(wns_values) >= 2:
            log.wns_before_phys_opt = wns_values[0]
            log.wns_after_phys_opt = wns_values[-1]
        elif len(wns_values) == 1:
            log.wns_after_phys_opt = wns_values[0]

        # 提取运行时间
        log.duration_seconds = self._extract_duration(content)

        return log

    def _parse_full_log(self, content: str, findings: Findings):
        """解析包含多个阶段的完整 log"""
        # 按阶段标识符拆分
        sections = self._split_by_stage(content)

        for stage_name, section_content in sections.items():
            stage_log = self._parse_stage(section_content, stage_name)
            if stage_name == "synthesis":
                findings.synth_log = stage_log
            elif stage_name == "place":
                findings.place_log = stage_log
            elif stage_name == "route":
                findings.route_log = stage_log

    def _split_by_stage(self, content: str) -> dict:
        """按阶段标识符拆分完整 log"""
        sections = {}
        positions = []

        for stage, markers in self.STAGE_MARKERS.items():
            for marker in markers:
                for m in re.finditer(re.escape(marker), content):
                    positions.append((m.start(), stage))

        positions.sort()

        for i, (pos, stage) in enumerate(positions):
            end = positions[i + 1][0] if i + 1 < len(positions) else len(content)
            if stage not in sections:
                sections[stage] = content[pos:end]

        return sections

    def _extract_messages(self, content: str) -> List[LogMessage]:
        """提取所有 ERROR/CRITICAL WARNING/WARNING/INFO"""
        messages = []
        for m in self.MESSAGE_PATTERN.finditer(content):
            level = m.group(1)
            code = m.group(2) or ""
            text = m.group(3).strip()
            # 只保留有价值的消息（过滤 INFO 中的噪音）
            if level in ("ERROR", "CRITICAL WARNING", "WARNING"):
                messages.append(LogMessage(
                    level=level, code=code, text=text
                ))
        return messages

    def _extract_congestion(self, content: str) -> List[CongestionReport]:
        """提取拥塞报告"""
        reports = []
        for m in self.CONGESTION_PATTERN.finditer(content):
            reports.append(CongestionReport(
                level=int(m.group(1)),
                region=m.group(2),
            ))
        return reports

    def _extract_all_wns(self, content: str) -> List[float]:
        """提取所有 WNS 值（按出现顺序）"""
        return [
            float(m.group(1))
            for m in self.TIMING_SUMMARY_PATTERN.finditer(content)
        ]

    def _extract_duration(self, content: str) -> Optional[float]:
        """提取运行时间"""
        # Vivado 通常在日志末尾输出运行时间
        time_pattern = re.search(
            r"(?:real|elapsed|Time)\s*[:=]?\s*(\d+):(\d+)(?::(\d+))?",
            content, re.IGNORECASE
        )
        if time_pattern:
            minutes = int(time_pattern.group(1))
            seconds = int(time_pattern.group(2))
            return minutes * 60 + seconds
        return None
```

```python
import re
from pathlib import Path
from dataclasses import dataclass, field
from typing import List, Optional

from vivado_ai.models.finding import XDCData


@dataclass
class XDCCommand:
    """XDC 命令"""

    args: dict          # 命令参数
    line: int           # 行号

@dataclass
class XDCPort:
    """端口信息"""
    name: str
    direction: str      # "input" | "output" | "inout"
    file_path: str = ""


class XDCParser:
    """
    解析 XDC 约束文件

    提取:
    - create_clock / create_generated_clock
    - set_input_delay / set_output_delay
    - set_clock_groups / set_false_path
    - set_max_delay / set_min_delay
    - set_multicycle_path
    - set_case_analysis / set_disable_timing
    - set_clock_uncertainty
    - 物理约束 (LOC, Pblock)
    """

    # XDC 命令匹配
    COMMAND_TYPES = [
        "create_clock",
        "create_generated_clock",
        "set_input_delay",
        "set_output_delay",
        "set_clock_groups",
        "set_false_path",
        "set_max_delay",
        "set_min_delay",
        "set_multicycle_path",
        "set_case_analysis",
        "set_disable_timing",
        "set_clock_uncertainty",
        "set_property",
        "set_bus_skew",
    ]

    def parse(self, file_path: Path) -> XDCData:
        """解析 XDC 文件"""
        content = file_path.read_text(encoding="utf-8", errors="ignore")
        commands = []

        for line_no, line in enumerate(content.splitlines(), 1):
            stripped = line.strip()
            if not stripped or stripped.startswith("#"):
                continue

            for cmd_type in self.COMMAND_TYPES:
                if stripped.startswith(cmd_type):
                    commands.append(XDCCommand(
                        type=cmd_type,
                        args=self._parse_args(stripped),
                        line=line_no,
                        file_path=str(file_path),
                        raw=stripped,
                    ))
                    break

        return XDCData(
            commands=commands,
            file_path=str(file_path),
        )

    def _parse_args(self, line: str) -> dict:
        """简化的 XDC 参数解析"""
        args = {}

        # 提取 -name 值
        name_match = re.search(r'-name\s+(\S+)', line)
        if name_match:
            args["name"] = name_match.group(1).strip('"')

        # 提取 -period 值
        period_match = re.search(r'-period\s+([\d.]+)', line)
        if period_match:
            args["period"] = float(period_match.group(1))

        # 提取 -clock 值
        clock_match = re.search(r'-clock\s+(?:\[(.*?)\]|(\S+))', line)
        if clock_match:
            args["clock"] = (
                clock_match.group(1) or clock_match.group(2)
            ).strip('"')

        # 提取 get_ports / get_clocks / get_cells 目标
        target_match = re.search(
            r'get_ports\s+(?:-of_objects\s+\S+\s+)?(\[\S+\]|\S+)',
            line
        )
        if target_match:
            args["port"] = target_match.group(1).strip('[]')

        # 提取 -asynchronous / -logically_exclusive / -physically_exclusive
        for flag in ("asynchronous", "logically_exclusive",
                     "physically_exclusive", "datapath_only"):
            if f"-{flag}" in line:
                args[flag] = True

        # 提取 -group 值（可能有多个）
        groups = re.findall(r'-group\s+(?:\[(.*?)\]|(\S+))', line)
        if groups:
            args["groups"] = [
                g[0] or g[1] for g in groups
            ]

        return args
```

```python
# models/report.py

from dataclasses import dataclass, field
from typing import List, Dict
from vivado_ai.models.issue import Issue, Severity


@dataclass
class CheckReport:
    """检查报告"""
    # 元信息
    mode: str              # "lint" | "check" | "analyze"
    total_rules: int       # 执行的规则总数
    total_issues: int      # 发现的问题总数

    # 分类统计
    by_severity: Dict[str, int] = field(default_factory=dict)
    by_group: Dict[str, List[Issue]] = field(default_factory=dict)

    # 评分
    score: int = 100       # 合规评分 0-100

    # 问题列表
    issues: List[Issue] = field(default_factory=list)

    # 优先修复建议
    priority_actions: List[str] = field(default_factory=list)

    def to_markdown(self) -> str:
        """生成 Markdown 格式报告"""
        lines = [
            "# Vivado Methodology Check Report",
            f"\n**Mode**: {self.mode}",
            f"**Score**: {self.score}/100",
            f"**Total Issues**: {self.total_issues}",
            "",
            "## Summary",
            "",
        ]
        for sev, count in self.by_severity.items():
            lines.append(f"- {sev}: {count}")

        lines.append("\n## Issues\n")
        for issue in self.issues:
            if issue.severity == Severity.PASS:
                continue
            lines.append(
                f"### [{issue.severity.value}] "
                f"{issue.rule_id}: {issue.rule_name}"
            )
            lines.append(f"\n{issue.message}")
            if issue.detail:
                lines.append(f"\n**Detail**: {issue.detail}")
            if issue.fix_suggestion:
                lines.append(f"\n**Fix**: {issue.fix_suggestion}")
            if issue.ug949_ref:
                lines.append(f"\n**Ref**: UG949 {issue.ug949_ref}")
            if issue.ai_explanation:
                lines.append(f"\n**AI Analysis**: {issue.ai_explanation}")
            lines.append("\n---\n")

        return "\n".join(lines)

    def to_json(self) -> dict:
        """生成 JSON 格式（供 API 使用）"""
        return {
            "mode": self.mode,
            "score": self.score,
            "total_issues": self.total_issues,
            "by_severity": self.by_severity,
            "issues": [
                {
                    "rule_id": i.rule_id,
                    "severity": i.severity.value,
                    "message": i.message,
                    "detail": i.detail,
                    "fix": i.fix_suggestion,
                    "location": i.location,
                    "ref": i.ug949_ref,
                }
                for i in self.issues
            ],
        }
```
